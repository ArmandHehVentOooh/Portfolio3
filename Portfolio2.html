    #c'est explicite
    $stock_de_PDF = "C:\Users\AVENTADOUR\Documents\Dossier de Quittances TEST"
    
    # Récupérer tous les PDF du dossier et de ses sous-dossiers
    $dossiers = Get-ChildItem -Path $stock_de_PDF -Filter "FactureBidon_*_*_MonENTREPRISE_*.pdf" -Recurse -ErrorAction SilentlyContinue -Force
    
    # Abréviations des mois
    $moisAbr = @{
        "janvier" = "janv."
        "février" = "févr."
        "mars" = "mars"
        "avril" = "avr."
        "mai" = "mai"
        "juin" = "juin"
        "juillet" = "juil."
        "août" = "août"
        "septembre" = "sept."
        "octobre" = "oct."
        "novembre" = "nov."
        "décembre" = "déc."
    }
    
    # Numéros des mois
    $moisNum = @{
        "janvier" = "01"
        "février" = "02"
        "mars" = "03"
        "avril" = "04"
        "mai" = "05"
        "juin" = "06"
        "juillet" = "07"
        "août" = "08"
        "septembre" = "09"
        "octobre" = "10"
        "novembre" = "11"
        "décembre" = "12"
    }
    
    # Parcourir chaque fichier PDF
    foreach ($dossier in $dossiers) {
        # Extraire les informations du nom de fichier
        if ($dossier.Name -match "FactureBidon_(\d{4})_(\w+)_MonENTREPRISE_(\d+)") {
            $annee = $matches[1]#c'est (\d{4}), 4 chifres
            $mois = $matches[2]#c'est (\w+), n lettres
            $numero = $matches[3]#c'est (\d+), n chiffres
    
            # Affichage pour le terminal
            Write-Host "Fichier: $($dossier.Name)"
            Write-Host "Année: $annee"
            Write-Host "Mois: $mois"
            Write-Host "Numéro: $numero"
    
            # laisser les deux derniers chiffres de l'année
            $anneeAbr = $annee.Substring(2, 2)
    
            # Vérifier si le mois existe dans la liste
            if ($moisNum.ContainsKey($mois) -and $moisAbr.ContainsKey($mois)) {
                # Obtenir le numéro du mois
                $moisNumFormate = $moisNum[$mois]
    
                # Obtenir l'abréviation du mois
                $moisAbrFormate = $moisAbr[$mois]
    
                # Créer le nom du dossier
                $dossierNom = "$anneeAbr$moisNumFormate QUIT $moisAbrFormate $annee"
    
                # Chemin complet du dossier de destination
                $dossierDestination = Join-Path -Path $stock_de_PDF -ChildPath $dossierNom
    
    
                # Créer le dossier s'il n'existe pas
                if (-not (Test-Path -Path $dossierDestination)) {
                    New-Item -ItemType Directory -Path $dossierDestination
                } else {
                    Write-Host "Le dossier $dossierDestination existe déjà."
                }
    
                # copier le fichier dans le dossier de destination
                Copy-Item -Path $dossier.FullName -Destination $dossierDestination
            } else {
                Write-Host "Le mois '$mois' n'existe pas on dirait."
            }
        } else {
            Write-Host "Le fichier $($dossier.Name) ne correspond pas au modèle attendu."
        }
    }
